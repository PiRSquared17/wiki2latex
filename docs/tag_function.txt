/*
Register this function as usual to the Mediawikiparser:
$wgParser->setHook("noindent", "NoIndent");

And put this line into your w2lConfig.php-file:
$w2l_tags['noindent'] = "NoIndent";

The W2L-Parser submits an additional value ($mode), so a function can decide,
what output should be returned.
*/
function NoIndent($input, $argv, &$parser, $mode = 'wiki') {
	switch($mode) {
		case 'latex':
			return '{\noindent}';
		break;
		case 'wiki':
			return '';
		break;
		default:
			return '';
		break;
	}  
}
